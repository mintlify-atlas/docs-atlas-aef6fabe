## Key Metrics

The reports component tracks the following metrics:

### Sales Period Data

```typescript
export interface SalesPeriodData {
```


```typescript
export interface CategorySales {
```

### Future Backend Integration

To connect to a real backend service:

```typescript
import { ReportsService } from '../../services/reports.service';

export class ReportsComponent implements OnInit {
  constructor(
    private reportsService: ReportsService,
    private messageService: MessageService
  ) {}
  
  ngOnInit() {
    this.loadSalesData();
  }
  
  loadSalesData() {
    this.reportsService.getSalesData(this.selectedPeriod).subscribe({
      next: (res) => {
        if (res.code === 0) {
          this.salesData = res.data;
        }
      },
      error: () => {
        this.messageService.add({
          severity: 'error',
          summary: 'Error',
          detail: 'No se pudieron cargar los datos de ventas.'
        });
      }
    });
  }
  
  onPeriodChange() {
    this.loadSalesData();
  }
}
```

## Metric Calculations

### Average Ticket Size

```typescript
calculateAvgTicket(sales: number, transactions: number): number {
  return transactions > 0 ? sales / transactions : 0;
}
```

### Growth Percentage

```typescript
calculateGrowth(current: number, previous: number): number {
  if (previous === 0) return 0;
  return ((current - previous) / previous) * 100;
}
```

### Category Percentage

```typescript
calculateCategoryPercentage(categoryAmount: number, totalSales: number): number {
  return totalSales > 0 ? (categoryAmount / totalSales) * 100 : 0;
}
```

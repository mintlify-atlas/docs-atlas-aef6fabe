# Inventory Management

The Inventory Management module provides complete control over your product catalog, stock levels across multiple warehouses, pricing, and categorization.

## Overview

Built with Angular 19 and PrimeNG Table component, the inventory system supports:
- Multi-warehouse stock management
- Min/max stock levels per warehouse
- Bulk import/export via Excel
- Category and unit of measure organization
- Product images with Digital Ocean integration
- Real-time filtering and search

## Statistics Calculation

The system automatically calculates inventory metrics:

```typescript
calculateTotals() {
    this.totalItems = 0;
    this.totalValue = 0;
    this.lowStockItems = [];
    
    this.inventory.forEach((item) => {
        const stockTotal = this.getProductTotalStock(item);
        this.totalItems += stockTotal;
        this.totalValue += stockTotal * (item.price || 0);
        
        // Low stock detection (below 5 units)
        if (item.manageStock && stockTotal <= 5) {
            this.lowStockItems.push(item);
        }
    });
}

getProductTotalStock(product: Product): number {
    if (!product.existencias) return 0;
    return product.existencias.reduce(
        (acc, curr) => acc + (curr.stock || 0), 0
    );
}
```

## Filtering Products

Advanced filtering by multiple criteria:

```typescript
filters = {
    nameOrReference: '',
    category: null,
    extension1: '',
    extension2: '',
    unitMeasure: '',
    active: true,
    minPrice: null,
    maxPrice: null
};

filterProducts() {
    this.filteredInventory = this.inventory.filter((item) => {
        const matchesSearch = !this.filters.nameOrReference ||
            item.name.toLowerCase().includes(
                this.filters.nameOrReference.toLowerCase()
            ) ||
            item.reference?.toLowerCase().includes(
                this.filters.nameOrReference.toLowerCase()
            ) ||
            item.barcode?.toLowerCase().includes(
                this.filters.nameOrReference.toLowerCase()
            );
        
        const matchesCategory = !this.filters.category || 
            item.categoryId === this.filters.category;
        
        const matchesExtension = !this.filters.extension1 || 
            item.extension1?.toLowerCase().includes(
                this.filters.extension1.toLowerCase()
            );
        
        const matchesActive = item.active === this.filters.active;
        
        return matchesSearch && matchesCategory && 
               matchesExtension && matchesActive;
    });
}
```

## Multi-Warehouse Management

### Warehouse UI Preparation

When adding or editing a product, warehouse data is unified into a single interface:

```typescript
interface WarehouseUI extends ProductWarehouseRequest {
    name: string;  // Display name in modal
}

prepareWarehouseUI(product?: Product) {
    this.warehousesUI = this.masterWarehouses.map((wh) => {
        // Find existing stock for this warehouse
        const stockRecord = product?.existencias?.find(
            (e) => e.wareHouseId === wh.id
        );
        
        // Find existing min/max for this warehouse
        const limitRecord = product?.maximosMinimos?.find(
            (m) => m.wareHouseId === wh.id
        );
        
        return {
            warehouseId: wh.id!,
            name: wh.name,
            stock: stockRecord?.stock ?? 0,
            minStock: limitRecord?.minStock ?? 0,
            maxStock: limitRecord?.maxStock ?? 0
        };
    });
}
```

### Image URL Generation

```typescript
getImageUrl(): string | null {
    if (!this.newProduct.imageUrl) return null;
    
    // Detect base64 encoded image
    const isBase64 = this.newProduct.imageUrl.startsWith('data:');
    
    if (isBase64) {
        return this.newProduct.imageUrl;
    }
    
    // Construct Digital Ocean URL
    return this.digitalOcean.url + '/' + this.newProduct.imageUrl;
}
```

## Bulk Import/Export

### Export to Excel

Export current filtered inventory with warehouse columns:

```typescript
exportToExcel() {
    const dataToExport = this.filteredInventory.map((p) => {
        const row: any = {
            ID: p.id,
            Código: p.code,
            Nombre: p.name,
            Referencia: p.reference || '',
            'Código Barras': p.barcode || '',
            'Categoría ID': p.categoryId,
            'Extension 1': p.extension1 || '',
            'Extension 2': p.extension2 || '',
            Precio: p.price,
            'Unidad Medida ID': p.unitMeasureId,
            'Maneja Stock': p.manageStock ? 'SI' : 'NO',
            Activo: p.active ? 'SI' : 'NO'
        };
        
        // Add dynamic columns for each warehouse
        this.masterWarehouses.forEach((wh) => {
            const stock = p.existencias?.find(
                (e) => e.wareHouseId === wh.id
            )?.stock || 0;
            const min = p.maximosMinimos?.find(
                (m) => m.wareHouseId === wh.id
            )?.minStock || 0;
            const max = p.maximosMinimos?.find(
                (m) => m.wareHouseId === wh.id
            )?.maxStock || 0;
            
            row[`Stock - ${wh.name}`] = stock;
            row[`Min - ${wh.name}`] = min;
            row[`Max - ${wh.name}`] = max;
        });
        
        return row;
    });
    
    const worksheet = XLSX.utils.json_to_sheet(dataToExport);
    const workbook = XLSX.utils.book_new();
    XLSX.utils.book_append_sheet(workbook, worksheet, 'Inventario');
    
    XLSX.writeFile(workbook, `Inventario_Compania_${this.companiaId}.xlsx`);
}
```

### Import from Excel

<Steps>
  <Step title="Select File">
    User selects an Excel file via file input:
    
    ```typescript
    importExcel(event: any) {
        this.loading = true;
        const file = event.target.files[0];
        if (!file) return;
        
        const reader = new FileReader();
        reader.onload = (e: any) => {
            const bstr = e.target.result;
            const wb = XLSX.read(bstr, { type: 'binary' });
            const wsname = wb.SheetNames[0];
            const ws = wb.Sheets[wsname];
            const data: any[] = XLSX.utils.sheet_to_json(ws);
            
            this.processBulkData(data);
        };
        reader.readAsBinaryString(file);
    }
    ```

  <Step title="Process Rows">
    Map Excel rows to ProductCreateRequest objects:

    ```typescript
    processBulkData(rows: any[]) {
        const requests: ProductCreateRequest[] = rows.map((row) => {
            // Map warehouse columns dynamically
            const warehousesForThisProduct = this.masterWarehouses.map((wh) => ({
                warehouseId: wh.id!,
                stock: Number(row[`Stock - ${wh.name}`] || 0),
                minStock: Number(row[`Min - ${wh.name}`] || 0),
                maxStock: Number(row[`Max - ${wh.name}`] || 0)
            }));
            
            return {
                id: Number(row['ID'] || 0),  // 0 = new product
                userId: this.userId,
                companiaId: this.companiaId,
                name: row['Nombre'],
                description: row['Descripción'] || '',
                barcode: String(row['Código Barras'] || ''),
                reference: String(row['Referencia'] || ''),
                extension1: row['Extension 1'] || '',
                extension2: row['Extension 2'] || '',
                price: Number(row['Precio'] || 0),
                imageUrl: row['Imagen URL'] || '',
                categoryId: Number(row['Categoría ID']),
                unitMeasureId: Number(row['Unidad Medida ID']),
                active: row['Activo'] === 'SI',
                manageStock: row['Maneja Stock'] === 'SI',
                warehouses: warehousesForThisProduct
            };
        });
        
        // Send bulk request
        this.inventoryService.createProductBulk(requests).subscribe({
            next: () => {
                this.loading = false;
                this.messageService.add({
                    severity: 'success',
                    summary: 'Carga Masiva',
                    detail: `Se procesaron ${requests.length} productos exitosamente.`
                });
                this.loadInventory();
            },
            error: (err) => {
                this.loading = false;
                this.messageService.add({
                    severity: 'error',
                    summary: 'Error',
                    detail: `No se pudo cargar el archivo: ${err.message}`
                });
            }
        });
    }
    ```

## Best Practices

<Steps>
  <Step title="Set Min/Max Stock Levels">
    Always configure minimum and maximum stock levels for products with `manageStock` enabled. This enables low-stock alerts.
  </Step>
  
  <Step title="Use Consistent Categories">
    Create and use categories consistently to enable better filtering and reporting.
  </Step>
  
  <Step title="Test Bulk Import First">
    Export your current inventory, modify it, and reimport a small subset first to verify the format before bulk uploads.
  </Step>
  
  <Step title="Leverage Custom Fields">
    Use `extension1` and `extension2` fields for product attributes like color, size, or brand for enhanced filtering.
  </Step>
</Steps>

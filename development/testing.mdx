## Overview

- **Jasmine** - Testing framework

## Testing Configuration

### Karma Configuration

The test runner is configured through Angular's build system:

**Location**: `angular.json`

```json
{
    "test": {
        "builder": "@angular-devkit/build-angular:karma",
        "options": {
            "polyfills": [
                "zone.js",
                "zone.js/testing"

            "tsConfig": "tsconfig.spec.json",
            "inlineStyleLanguage": "scss",

                    "glob": "**/*",
                    "input": "public"
```

## Running Tests

### Execute All Tests

```bash
npm test
```

This runs `ng test` which:
- Compiles the application
- Launches Karma test runner
- Opens Chrome browser
- Runs all `*.spec.ts` files
- Watches for file changes

### Single Run (CI/CD)

```bash
ng test --watch=false --browsers=ChromeHeadless
```

### Code Coverage

```bash
ng test --code-coverage
```

Coverage reports are generated in the `coverage/` directory.

### Testing Methods

```typescript
import { of, throwError } from 'rxjs';

describe('AgentsComponent Methods', () => {
    it('should load agents successfully', () => {
        const mockResponse = {
            code: 0,
            data: [
                { id: 1, name: 'Agent 1' },
                { id: 2, name: 'Agent 2' }
            ]
        };

        whatsappService.getAgents.and.returnValue(of(mockResponse));
        component.selectedConfigId = 1;

        component.fetchAssignedAgents();

        expect(component.assignedAgents.length).toBe(2);
        expect(component.isLoading).toBeFalse();
    });

    it('should handle error when loading agents fails', () => {
        whatsappService.getAgents.and.returnValue(
            throwError(() => new Error('API Error'))
        );
        component.selectedConfigId = 1;

        component.fetchAssignedAgents();

        expect(component.assignedAgents).toEqual([]);
        expect(component.isLoading).toBeFalse();
    });
});
```

## Testing Services

### Service Test Example

```typescript
import { TestBed } from '@angular/core/testing';
import { HttpClientTestingModule, HttpTestingController } from '@angular/common/http/testing';
import { MasterService } from './master.service';
import { User } from '../api/permissions';

describe('MasterService', () => {
    let service: MasterService;
    let httpMock: HttpTestingController;

    beforeEach(() => {
        TestBed.configureTestingModule({
            imports: [HttpClientTestingModule],
            providers: [MasterService]
        });

        service = TestBed.inject(MasterService);
        httpMock = TestBed.inject(HttpTestingController);
    });

    afterEach(() => {
        httpMock.verify();
    });

    it('should be created', () => {
        expect(service).toBeTruthy();
    });

    it('should fetch users', () => {
        const mockUsers: User[] = [
            { userId: 1, username: 'user1', email: 'user1@example.com' },
            { userId: 2, username: 'user2', email: 'user2@example.com' }
        ];
        const companiaId = 1;

        service.getUsers(companiaId).subscribe(users => {
            expect(users.length).toBe(2);
            expect(users).toEqual(mockUsers);
        });

        const req = httpMock.expectOne(`${service.apiUrl}/users/${companiaId}`);
        expect(req.request.method).toBe('GET');
        req.flush(mockUsers);
    });

    it('should handle error when fetching users fails', () => {
        const companiaId = 1;
        const errorMessage = 'Failed to load users';

        service.getUsers(companiaId).subscribe(
            () => fail('should have failed'),
            (error) => {
                expect(error.status).toBe(500);
            }
        );

        const req = httpMock.expectOne(`${service.apiUrl}/users/${companiaId}`);
        req.flush(errorMessage, { status: 500, statusText: 'Server Error' });
    });
});
```

## Testing Permissions

```typescript
describe('Permissions', () => {
    beforeEach(() => {
        component.permissions = [
            {
                module: 'Agentes',
                canView: true,
                canCreate: true,
                canEdit: false,
                canDelete: false,
                canExport: true
            }
        ];
    });

    it('should apply permissions correctly', () => {
        component['applyPermissions']();

        expect(component.canView).toBeTrue();
        expect(component.canCreate).toBeTrue();
        expect(component.canEdit).toBeFalse();
        expect(component.canDelete).toBeFalse();
        expect(component.canExport).toBeTrue();

        expect(component.canCreate).toBeFalse();
    });
});
```

## Jasmine Matchers

### Common Matchers

```typescript
// Equality
expect(value).toBe(expected);           // ===
expect(value).toEqual(expected);        // Deep equality
expect(value).toBeTruthy();             // Truthy
expect(value).toBeFalsy();              // Falsy
expect(value).toBeNull();               // null
expect(value).toBeUndefined();          // undefined
expect(value).toBeDefined();            // defined

// Numerical
expect(value).toBeGreaterThan(3);
expect(value).toBeLessThan(5);
expect(value).toBeCloseTo(4.2, 1);      // Precision

// Strings
expect(string).toContain('substring');
expect(string).toMatch(/pattern/);

// Arrays
expect(array).toContain(item);
expect(array.length).toBe(5);

// Objects
expect(obj).toHaveProperty('key');
expect(obj.key).toBe('value');

// Functions
expect(fn).toThrow();
expect(fn).toHaveBeenCalled();
expect(fn).toHaveBeenCalledWith(arg1, arg2);
expect(fn).toHaveBeenCalledTimes(3);
```

## Best Practices

<Tip>
**Testing Guidelines**
- Write tests for all components and services
- Test both success and error scenarios
- Use descriptive test names
- Keep tests isolated and independent
- Mock external dependencies
- Test user interactions
- Verify permissions are checked
</Tip>

<Warning>
**Common Testing Mistakes**
- Don't test framework code (Angular internals)
- Don't forget to call `fixture.detectChanges()`
- Don't use real HTTP calls - use mocks
- Don't test implementation details
- Don't forget to verify HTTP mocks with `httpMock.verify()`
</Warning>

## Debugging Tests

### Chrome DevTools

1. Run `ng test`
2. Click "DEBUG" button in Karma browser
3. Open Chrome DevTools (F12)
4. Set breakpoints in the Sources tab
5. Refresh the page to hit breakpoints
